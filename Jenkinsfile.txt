pipeline{
    
    agent any 
    
    stages {
        
        stage('Git Checkout'){
            
            steps{
                
                script{
                    
checkout([$class: 'GitSCM', branches: [[name: '*/repo-abdessalem']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/dhiabenmaati/esprit-devops-repo.git']]])
                
            }
        
            }
            
        }
            
               stage('Cleaning the project') {
            steps{
                sh "mvn -B -DskipTests clean  " 
            }
        }
        
        
        stage('Unit Tests') {
            steps{
                sh "mvn test " 
            }
        }
        
        
        
        stage('Artifact Construction') {
            steps{
                sh "mvn -B -DskipTests package " 
            }
        }
        }
}
